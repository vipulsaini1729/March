        AvailableQuantity = Integer.valueOf(model.getValueAt(myIdex,2).toString());
        ProductName.setText(model.getValueAt(myIdex,1).toString());
        ProductPrice = Double.valueOf(model.getValueAt(myIdex,3).toString()); 
        NewQuantity= AvailableQuantity - Integer.parseInt(ProductQuantity.getText());
        ProductTotal = ProductPrice * Double.valueOf(ProductQuantity.getText());

    }                                         
    
     public void selectSeller(){
      try{
        Connection con;
        PreparedStatement pre;
        int cc;
        Class.forName("com.mysql.jdbc.Driver");
        con = (Connection) DriverManager.getConnection("jdbc:mysql://localhost/managementsystem","root","Shruti@23"); 
        java.sql.Statement s = con.createStatement();
        pre= con.prepareStatement("select * from producttable");
        ResultSet rs = pre.executeQuery();
        ResultSetMetaData RSMD = rs.getMetaData();
        cc = RSMD.getColumnCount();
        DefaultTableModel DFT = (DefaultTableModel) ProductTable.getModel();
       DFT.setRowCount(0);
      while (rs.next()) {
         Vector v2=new Vector();
       for (int j=1;j <cc;j++){
             v2.add(rs.getString("ProductID"));
             v2.add(rs.getString("ProductNmae"));
             v2.add(rs.getString("ProductQantity"));
             v2.add(rs.getString("ProductPrice"));
            v2.add(rs.getString("ProductCategory"));
           }
            DFT.addRow(v2);
      }
      }
     catch (ClassNotFoundException | SQLException ex){
                Logger.getLogger(Product.class.getName()).log(Level.SEVERE, null, ex);
            }
}

    private void AddToBillButtonMouseClicked(java.awt.event.MouseEvent evt) {                                             
        // TODO add your handling code here:
        if(ProductQuantity.getText().isEmpty() || ProductName.getText().isEmpty())
        { 
            JOptionPane.showMessageDialog(this,"Missing Information");
        } 
       else if (AvailableQuantity < Integer.parseInt(ProductQuantity.getText())){
             JOptionPane.showMessageDialog(this,"not Enough in stock");
      } else{
            i++;
         ProductTotal = ProductPrice * Double.valueOf(ProductQuantity.getText());
        GrandTotal = GrandTotal + ProductTotal;
       if(i==1){
           BillText.setText(BillText.getText()+".............Family point.......\n"+ "NUM    PRODUCT  PRICE  QUANTITY  TOTAL \n"+i+"      " + ProductName +"  " +ProductQuantity.getText() + "     " +ProductTotal+"\n");
       }else{
            BillText.setText(BillText.getText()+ i +"      "+ProductName.getText()+"    "+ProductPrice+"     "+ ProductQuantity.getText() +"      " +ProductTotal+"\n");
         }
         TotalLabel.setText("Rs."+GrandTotal);
         update();
              }  
    }
